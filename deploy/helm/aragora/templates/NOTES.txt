To access Aragora, set up port forwarding:

  # Backend API
  kubectl port-forward svc/{{ include "aragora.fullname" . }}-backend {{ .Values.backend.service.port }}:{{ .Values.backend.service.port }} -n {{ .Release.Namespace }}

  {{- if .Values.frontend.enabled }}
  # Frontend dashboard
  kubectl port-forward svc/{{ include "aragora.fullname" . }}-frontend {{ .Values.frontend.service.port }}:{{ .Values.frontend.service.port }} -n {{ .Release.Namespace }}
  {{- end }}

  Then open:
    API:        http://localhost:{{ .Values.backend.service.port }}/api
    Health:     http://localhost:{{ .Values.backend.service.port }}/healthz
  {{- if .Values.frontend.enabled }}
    Dashboard:  http://localhost:{{ .Values.frontend.service.port }}
  {{- end }}

{{- end }}

HEALTH CHECK:

  kubectl get pods -l app.kubernetes.io/instance={{ .Release.Name }} -n {{ .Release.Namespace }}
